data VPCalcAddress = VPCalcAddress String Int deriving (Eq, Generic, Show)

type vPCalcAddress =
  | VPCalcAddress of string * int

let encodeVPCalcAddress x =
  match x with
  | VPCalcAddress (y0,y1) ->
     Aeson.Encode.array [| Aeson.Encode.string y0 ; Aeson.Encode.int y1 |]

let decodeVPCalcAddress json =
  match Aeson.Decode.string
  ; int json with
  | v -> Js_result.Ok (VPCalcAddress v)
  | exception Aeson.Decode.DecodeError msg -> Js_result.Error ("decodeVPCalcAddress: " ^ msg)
